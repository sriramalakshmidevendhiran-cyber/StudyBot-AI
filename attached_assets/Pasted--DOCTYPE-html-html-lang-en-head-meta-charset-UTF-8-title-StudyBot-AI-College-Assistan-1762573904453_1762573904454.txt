<script>
async function askTutor(){
  const topic=document.getElementById("topic").value;
  const res=await fetch("/tutor",{method:"POST",headers:{"Content-Type":"application/json"},
  body:JSON.stringify({topic})});
  const data=await res.json();
  document.getElementById("output").textContent=data.result;
  speak(data.result);
}
async function getQuiz(){
  const topic=document.getElementById("topic").value;
  const res=await fetch("/quiz",{method:"POST",headers:{"Content-Type":"application/json"},
  body:JSON.stringify({topic})});
  const data=await res.json();
  document.getElementById("output").textContent=data.result;
  speak("Here is your quiz on "+topic);
}
async function getPlan(){
  const topic=document.getElementById("topic").value;
  const res=await fetch("/plan",{method:"POST",headers:{"Content-Type":"application/json"},
  body:JSON.stringify({subject:topic,days:5})});
  const data=await res.json();
  document.getElementById("output").textContent=data.result;
  speak("Here is your five-day plan for "+topic);
}

 document.getElementById("topic").value=e.results[0][0].transcript;
